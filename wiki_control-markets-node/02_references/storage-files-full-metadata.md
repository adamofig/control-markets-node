So far this is my best model 

interface IStorageAsset {
  _id?: string;
  id: string;
  name: string;
  description?: string;
  type: 'image' | 'audio' | 'video';
  category: 'upload' | 'generated' | 'source'; // discriminator
  storage: CloudFileStorage;   // the actual file
  thumbnail?: CloudFileStorage;
  status?: 'pending' | 'processing' | 'completed' | 'failed';
  tag?: string;
  auditable?: IAuditable;

  generationMetadata?: {
    operationId?: string;
    prompt?: string;
    model?: string;
    provider?: 'vertex' | 'comfy' | string;
    workflow?: 'text-to-video' | 'image-to-video' | 'image-audio-to-video';
    request?: any;
    response?: any;
  };


  resourceMetadata?: {
    sourceUrl?: string;
    content?: string;
    contentEnhancedAI?: string;
    transcription?: any;
    captions?: { remotion?: any[]; tiktokStyle?: any[] };
    frames?: CloudFileStorage[];
    separatedAudio?: { vocals?: CloudFileStorage; no_vocals?: CloudFileStorage };
  };
}
